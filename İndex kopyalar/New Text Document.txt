function azNow(){ return new Date(); }
function todayStr(){ const n=azNow(),pad=x=>String(x).padStart(2,'0'); return `${n.getFullYear()}-${pad(n.getMonth()+1)}-${pad(n.getDate())}`; }

function azDateStr(iso){ const d=new Date(iso); return d.toLocaleDateString()+ ' ' + d.toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'}); }
function azDateOnly(iso){ return iso.slice(0,10); }
function inRange(dateStr, from, to){
  if(!from && !to) return true;
  if(from && dateStr < from) return false;
  if(to   && dateStr > to)   return false;
  return true;
}

function doRestore(ev){
  const file=ev.target.files[0]; if(!file)return;
  if(!confirm(t('restore_confirm')))return;
  const reader=new FileReader();
  reader.onload=e=>{
    try{
      const data=JSON.parse(e.target.result);
      if(data.rr_menu) localStorage.setItem('rr_menu',JSON.stringify(data.rr_menu));
      if(data.rr_sales) localStorage.setItem('rr_sales',JSON.stringify(data.rr_sales));
      if(data.rr_expenses) localStorage.setItem('rr_expenses',JSON.stringify(data.rr_expenses));
      if(data.rr_incomes) localStorage.setItem('rr_incomes',JSON.stringify(data.rr_incomes));
      if(data.rr_kassa) localStorage.setItem('rr_kassa',JSON.stringify(data.rr_kassa));
      if(data.rr_dyn_users) localStorage.setItem('rr_dyn_users',JSON.stringify(data.rr_dyn_users));
      if(data.rr_sys_users) localStorage.setItem('rr_sys_users',JSON.stringify(data.rr_sys_users));
      if(data.rr_settings) localStorage.setItem('rr_settings',JSON.stringify(data.rr_settings));
      if(data.rr_lang){ lang=data.rr_lang; localStorage.setItem('rr_lang',lang); }
      alert(t('restore_done'));
      location.reload();
    }catch(err){ alert('Xəta: düzgün backup faylı deyil'); }
  };

  ev.target.value='';
}

const MAX_ATTEMPTS = 5;
const LOCKOUT_MS = 5 * 60 * 1000; // 5 dəqiqə

function getLoginAttempts(){
  const s = sessionStorage.getItem('rr_login_att');
  return s ? JSON.parse(s) : {count:0, lockedUntil:0};
}
function saveLoginAttempts(a){ sessionStorage.setItem('rr_login_att', JSON.stringify(a)); }
function resetLoginAttempts(){ sessionStorage.removeItem('rr_login_att'); }

function doLogin(){
  const u=document.getElementById('loginUser').value.trim();
  const p=document.getElementById('loginPass').value;
  const err=document.getElementById('loginErr');

  // Brute force yoxlaması
  const att = getLoginAttempts();
  const now = Date.now();
  if(att.lockedUntil && now < att.lockedUntil){
    const remaining = Math.ceil((att.lockedUntil - now)/1000);
    err.textContent = `❌ Hesab müvəqqəti kilidlənib. ${remaining} saniyə gözləyin.`;
    return;
  }

  // Input sanitize — yalnız icazə verilən simvollar
  if(!/^[a-zA-Z0-9_.@-]{1,40}$/.test(u)){
    err.textContent = t('err_user');
    return;
  }
  if(p.length < 1 || p.length > 100){
    err.textContent = t('err_pass');
    return;
  }

  const all=getAllUsers();
  const userExists = !!all[u];
  const passMatch = userExists && all[u].pass === p;

  if(!userExists || !passMatch){
    // Uğursuz cəhd sayını artır
    att.count = (att.count||0) + 1;
    if(att.count >= MAX_ATTEMPTS){
      att.lockedUntil = now + LOCKOUT_MS;
      att.count = 0;
      err.textContent = `❌ Çox sayda yanlış cəhd. 5 dəqiqə gözləyin.`;
      saveLoginAttempts(att);
      return;
    }
    const left = MAX_ATTEMPTS - att.count;
    err.textContent = (userExists ? t('err_pass') : t('err_user')) + ` (${left} cəhd qalır)`;
    saveLoginAttempts(att);
    return;
  }

  if(selectedRole==='admin'&&all[u].role!=='admin'){err.textContent=t('err_role_admin');return;}
  if(selectedRole==='seller'&&all[u].role!=='seller'){err.textContent=t('err_role_seller');return;}

  // Uğurlu giriş — sayacı sıfırla
  resetLoginAttempts();
  currentUser={username:u,role:all[u].role,fullname:all[u].fullname,label:all[u].role==='admin'?'Admin':'Satıcı — '+all[u].fullname};
  err.textContent='';
  startApp();
}

function calcT(s){
  const subtotal=s.cart.reduce((x,c)=>x+c.price*c.qty,0);
  let discAmt=0;
  if(s.dactive&&s.dval>0) discAmt=s.dtype==='percent'?subtotal*(s.dval/100):Math.min(s.dval,subtotal);
  const deliveryAmt=(s.delivery&&!s.deliveryFree&&s.deliveryCost>0)?s.deliveryCost:0;
  return{subtotal,discAmt,deliveryAmt,total:Math.max(0,subtotal-discAmt)+deliveryAmt};
}
function addCart(id){
  const s=aS();const item=getMenu().find(m=>m.id===id);if(!item)return;
  const ex=s.cart.find(c=>c.id===id);
  ex?ex.qty++:s.cart.push({id,name:item.name,price:item.price,cost:item.cost||0,qty:1});

function confirmSale(){
  const s=aS();if(!s.cart.length)return;
  // Məbləğ manipulyasiyasına qarşı yoxlama
  const menu=getMenu();
  for(const ci of s.cart){
    const menuItem = menu.find(m=>m.id===ci.id);
    if(!menuItem){alert('Səbətdə silinmiş məhsul var. Yeniləyin.');s.cart=[];renderSale();return;}
    if(Math.abs(ci.price - menuItem.price) > 0.001){
      alert('Qiymət dəyişikliyi aşkarlandı. Səbət yeniləndi.');
      ci.price = menuItem.price; ci.cost = menuItem.cost||0;
      renderSale(); return;
    }
    if(ci.qty < 1 || ci.qty > 999 || !Number.isInteger(ci.qty)){alert('Yanlış miqdar.');return;}
  }
  if(s.dactive && s.dtype==='percent' && (s.dval<0||s.dval>100)){s.dactive=false;s.dval=0;}
  if(s.dactive && s.dtype==='amount' && s.dval<0){s.dactive=false;s.dval=0;}
  const customer=s.customer.trim()||'Anonim';
  const{subtotal,discAmt,deliveryAmt,total}=calcT(s);
  const sale={id:Date.now(),date:azISONow(),sessionName:s.name,customer,
    items:JSON.parse(JSON.stringify(s.cart)),subtotal,discAmt,dtype:s.dactive?s.dtype:null,dval:s.dactive?s.dval:0,
    delivery:s.delivery,deliveryFree:s.deliveryFree||false,deliveryCost:deliveryAmt,deliveryNote:s.deliveryNote||'',total,seller:currentUser.username,status:'Tamamlandı',note:''};
  const sales=getSales();sales.push(sale);saveSales(sales);
  kassaAdd('in',`Satış #${String(sale.id).slice(-6)} — ${s.name}`,total);
  s.cart=[];s.customer='';s.dactive=false;s.dval=0;s.delivery=false;s.deliveryFree=false;s.deliveryCost=0;s.deliveryNote='';
  renderBar();renderSale();showReceipt(sale);
}

function showReceipt(sale){

function setReportPeriod(p){
  const now=azNow();const pad=x=>String(x).padStart(2,'0');
  let from,to=todayStr();
  if(p==='today'){from=to;}
  else if(p==='week'){const d=new Date(now);d.setDate(d.getDate()-d.getDay()+1);from=`${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}`;}
  else if(p==='month'){from=`${now.getFullYear()}-${pad(now.getMonth()+1)}-01`;}
  document.getElementById('rFrom').value=from;
  document.getElementById('rTo').value=to;
  renderReports();
}
function renderReports(){
  const from=document.getElementById('rFrom').value;
  const to=document.getElementById('rTo').value;
  const sales=getSales().filter(s=>inRange(azDateOnly(s.date),from,to));
  const expenses=getExpenses().filter(e=>inRange(e.date,from,to));
  const incomes=getIncomes().filter(i=>inRange(i.date,from,to));
  const salesRev=sales.reduce((x,s)=>x+s.total,0);
  const costOfGoods=sales.reduce((x,s)=>x+s.items.reduce((a,it)=>a+(it.cost||0)*it.qty,0),0);
  const totalExp=expenses.reduce((x,e)=>x+e.amount,0);
  const totalInc=incomes.reduce((x,i)=>x+i.amount,0);
  const profit=salesRev+totalInc-costOfGoods-totalExp;
  document.getElementById('reportContent').innerHTML=`
    <div class="report-summary">
      <div class="report-block"><div class="rb-label">${t('report_count')}</div><div class="rb-val">${sales.length}</div></div>
      <div class="report-block"><div class="rb-label">${t('report_sales')}</div><div class="rb-val" style="color:var(--green)">${salesRev.toFixed(2)} ₼</div></div>
      <div class="report-block"><div class="rb-label">${t('report_cost')}</div><div class="rb-val" style="color:var(--accent2)">${costOfGoods.toFixed(2)} ₼</div></div>
      <div class="report-block"><div class="rb-label">${t('report_exp')}</div><div class="rb-val" style="color:var(--red)">${totalExp.toFixed(2)} ₼</div></div>
      <div class="report-block"><div class="rb-label">${t('report_inc')}</div><div class="rb-val" style="color:var(--blue)">${totalInc.toFixed(2)} ₼</div></div>

    </div>

      <div style="font-size:13px;font-weight:700;margin-bottom:14px;">📋 ${t('report_sales')} — detallar</div>
      <div style="overflow-x:auto;">
      <table class="sales-table">
        <thead><tr><th>${t('menu_cat')}</th><th>${t('menu_name')}</th><th>Say</th><th>${t('report_sales')}</th><th>${t('report_cost')}</th><th>Marja</th></tr></thead>
        <tbody>${renderReportByItem(sales)}</tbody>
      </table>
      </div>
    </div>`;
}
function renderReportByItem(sales){
  const map={};
  sales.forEach(s=>s.items.forEach(it=>{
    const k=`${it.id}`;
    if(!map[k]) map[k]={name:it.name,qty:0,rev:0,cost:0};
    map[k].qty+=it.qty; map[k].rev+=it.price*it.qty; map[k].cost+=(it.cost||0)*it.qty;
  }));
  const rows=Object.values(map).sort((a,b)=>b.rev-a.rev);
  if(!rows.length) return `<tr><td colspan="6" style="text-align:center;color:var(--muted);padding:20px">${t('no_data')}</td></tr>`;
  return rows.map(r=>`<tr>
    <td>${r.name}</td><td></td><td>${r.qty}</td>
    <td style="color:var(--green)">${r.rev.toFixed(2)} ₼</td>
    <td style="color:var(--accent2)">${r.cost.toFixed(2)} ₼</td>
    <td style="color:var(--purple)">${(r.rev-r.cost).toFixed(2)} ₼</td>
  </tr>`).join('');
}

  const totalIn=log.filter(e=>e.type==='in').reduce((x,e)=>x+e.amount,0);
  const totalOut=log.filter(e=>e.type==='out').reduce((x,e)=>x+e.amount,0);
  document.getElementById('kassaStats').innerHTML=
    `<div class="stat-card"><div class="stat-label">Cari Balans</div><div class="stat-value ${balance>=0?'c-green':'c-accent'}">${balance.toFixed(2)} ₼</div></div>`+
    `<div class="stat-card"><div class="stat-label">Ümumi Daxilolma</div><div class="stat-value c-green">${totalIn.toFixed(2)} ₼</div></div>`+
    `<div class="stat-card"><div class="stat-label">Ümumi Xərc/Çıxım</div><div class="stat-value c-accent">${totalOut.toFixed(2)} ₼</div></div>`;
  let running=0;
  const withBal=log.map(e=>{running+=(e.type==='in'?e.amount:-e.amount);return{...e,running};});
  const filtered=withBal.filter(e=>inRange(azDateOnly(e.date),from,to)).slice().reverse();
  const tb=document.getElementById('kassaBody');
  if(!filtered.length){tb.innerHTML=`<tr><td colspan="5" style="text-align:center;color:var(--muted);padding:24px">${t('no_data')}</td></tr>`;return;}
  tb.innerHTML=filtered.map(e=>`<tr>
    <td>${azDateStr(e.date)}</td>
    <td><span class="status-badge ${e.type==='in'?'kb-in':'kb-out'}">${e.type==='in'?'▲ Daxil':'▼ Çıxım'}</span></td>
    <td>${e.description}</td>
    <td style="font-weight:700;color:${e.type==='in'?'var(--green)':'var(--red)'}">${e.type==='in'?'+':'−'}${e.amount.toFixed(2)} ₼</td>
    <td style="color:${e.running>=0?'var(--text)':'var(--red)'}">${e.running.toFixed(2)} ₼</td>
  </tr>`).join('');
}

function saveUser(){

  const uname=document.getElementById('uUsername').value.trim().toLowerCase().replace(/\s+/g,'_');
  const pass=document.getElementById('uPassword').value;
  const role=document.getElementById('uRole').value;
  const perms=getFormPerms();
  if(!fullname){alert('Ad Soyad daxil edin');return;}
  if(!uname){alert('Login daxil edin');return;}
  if(!/^[a-z0-9_]+$/.test(uname)){alert('Login yalnız latın hərfləri, rəqəm və _ ola bilər');return;}
  const isSys=SYS_USER_KEYS.includes(uname);
  if(editingUsername){
    if(isSys){
      if(!pass){alert('Yeni parol daxil edin');return;}
      const su=getSysUsers();su[editingUsername]={...su[editingUsername],pass};saveSysUsers(su);
    }else{
      const dyn=getDynUsers();const ex=dyn[editingUsername];if(!ex)return;
      dyn[editingUsername]={...ex,fullname,role,perms,pass:pass?pass:ex.pass};saveDynUsers(dyn);
    }
    cancelEditUser();renderUserCards();alert('Yeniləndi!');
  }else{
    if(!pass){alert('Şifrə daxil edin');return;}
    if(getAllUsers()[uname]){alert('Bu login artıq mövcuddur');return;}
    const dyn=getDynUsers();dyn[uname]={fullname,pass,role,perms,sys:false};saveDynUsers(dyn);
    cancelEditUser();renderUserCards();alert('Əlavə edildi!');
  }
}

}

let editingSaleId = null;
function startEditSale(id){
  const sale = getSales().find(s=>s.id===id);
  if(!sale) return;
  editingSaleId = id;
  document.getElementById('seCustomer').value = sale.customer||'';
  document.getElementById('seSession').value = sale.sessionName||'';
  document.getElementById('seTotal').value = sale.total;
  document.getElementById('seDelivery').checked = !!sale.delivery;
  document.getElementById('seDeliveryFree').checked = !!sale.deliveryFree;
  document.getElementById('seDeliveryCost').value = sale.deliveryCost||'';
  document.getElementById('seNote').value = sale.note||'';
  document.getElementById('saleEditModal').classList.add('show');
}
function closeSaleEditModal(){
  editingSaleId = null;
  document.getElementById('saleEditModal').classList.remove('show');
}
function saveSaleEdit(){
  if(!editingSaleId) return;
  const sales = getSales();
  const idx = sales.findIndex(s=>s.id===editingSaleId);
  if(idx<0){ closeSaleEditModal(); return; }
  const oldTotal = sales[idx].total;
  const newTotal = parseFloat(document.getElementById('seTotal').value)||0;
  const delivery = document.getElementById('seDelivery').checked;
  const deliveryFree = document.getElementById('seDeliveryFree').checked;
  const deliveryCost = deliveryFree ? 0 : (parseFloat(document.getElementById('seDeliveryCost').value)||0);
  sales[idx] = {
    ...sales[idx],
    customer: document.getElementById('seCustomer').value.trim()||sales[idx].customer,
    sessionName: document.getElementById('seSession').value.trim()||sales[idx].sessionName,
    total: newTotal,
    delivery, deliveryFree, deliveryCost,
    note: document.getElementById('seNote').value.trim(),
    editedBy: currentUser.username,
    editedAt: azISONow(),
  };
  saveSales(sales);
  // Kassa balansını düzəlt — fərqi kompensasiya et
  if(newTotal !== oldTotal){
    const diff = newTotal - oldTotal;
    if(diff > 0) kassaAdd('in', `Satış düzəliş #${String(editingSaleId).slice(-6)} (+${diff.toFixed(2)})`, diff);
    else kassaAdd('out', `Satış düzəliş #${String(editingSaleId).slice(-6)} (${diff.toFixed(2)})`, Math.abs(diff));
  }
  closeSaleEditModal();
  renderHistory();
  alert('Satış yeniləndi!');
}
function deleteSale(id){
  if(!confirm('Bu satışı silmək istəyirsiniz? Kassa balansı da azalacaq.')) return;
  const sales = getSales();
  const sale = sales.find(s=>s.id===id);
  if(sale){
    kassaAdd('out', `Satış silindi #${String(id).slice(-6)} — ${sale.sessionName||''}`, sale.total);
  }
  saveSales(sales.filter(s=>s.id!==id));
  renderHistory();
}
